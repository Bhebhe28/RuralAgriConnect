Criterion 3 — Database & Query Optimization

  • Parameterized queries throughout — no SQL injection
  • Single DB instance shared across all routes
  • Explicit field selection — no SELECT * in production
  • JOINs used to avoid N+1 queries
  • 17 tables with foreign keys and cascade deletes
  • Seed file for reproducible demo data
// Explicit fields + JOIN — no N+1
SELECT u.user_id, u.full_name,
       r.role_name, f.region
FROM users u
LEFT JOIN roles r ON u.role_id = r.role_id
LEFT JOIN farmers f ON f.farmer_id = u.user_id
WHERE u.user_id = ?

// Never: SELECT * FROM users

Rate Limiting10 attempts per 15 min on /api/auth — brute force protection
🔑
bcrypt passwordsNever stored plain text — cost factor 10
// middleware/auth.ts
export function authenticate(req, res, next) {
  const token = req.headers
    .authorization?.split(' ')[1];
  if (!token) return res.status(401)...;
  req.user = jwt.verify(token, SECRET);
  next();
}

// Route usage:
router.post('/publish',
  authenticate, requireAdmin, handler);

/**
* formatDate — DD MMM YYYY display format.
* Used on advisory cards and report headers.
* Mirrors server dateUtils for consistency.
*/
export function formatDate(date: string): string {
  return new Date(date).toLocaleDateString(
    'en-ZA', { day:'2-digit',
      month:'short', year:'numeric' }
  );
}
